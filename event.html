// Hanersc Event Management Example
encod english
str: eventName = "Hanersc Developer Conference"
int: maxAttendees = 200
int: registeredAttendees = 0
bool: registrationOpen = true

system.out: "Welcome to " + eventName

cicle(registrationOpen) {
    system.out: "Available spots: "
    system.out: maxAttendees - registeredAttendees
    
    if(registeredAttendees >= maxAttendees) {
        registrationOpen = false
        system.out: "Registration is now closed!"
    }
    
    registeredAttendees = registeredAttendees + 1
}

system.out: "Event registration completed successfully!"
